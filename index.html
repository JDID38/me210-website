Autonomous Curling Robot

ME210: Introduction to Mechatronics — Stanford University

Project Overview

Our team designed and built an autonomous curling robot for the ME210 final competition. The robot uses a mecanum-wheel drivetrain for omnidirectional movement, an angled puck slide for consistent delivery, and a flywheel launcher to propel curling pucks down the playing surface toward the target zone.

The system is controlled by an Arduino Mega running a custom state-machine architecture with non-blocking timers, enabling the robot to autonomously navigate the arena, collect pucks, and execute precision shots within the two-minute match window.

This portfolio documents our mechanical design, electrical systems, software architecture, and the team behind the project.

Electrical Design

Schematics, wiring diagrams, and the stories behind our power system challenges.

Main Power Distribution

The robot runs on a 12 V LiPo battery pack with separate regulation paths for logic (5 V via buck converter) and motor power (direct 12 V through high-current MOSFETs). Schottky diodes prevent back-EMF from the flywheel motors from damaging the regulator.

During stall conditions, the flywheel motors draw upwards of 15 A per channel. We added inline fuses and thermal monitoring after an early prototype burned through a motor driver IC during a stall event in testing.

Arduino Mega Wiring

The Arduino Mega serves as the central controller, interfacing with four mecanum motor drivers via PWM, the flywheel ESC, limit switches for puck detection, and IR sensors for line following and arena navigation.

All signal lines are routed through a custom perfboard shield that sits on top of the Mega, keeping wiring organized and minimizing noise coupling between high-current motor lines and sensitive analog inputs.

High-Current Motor Driver Circuits

Each flywheel motor is driven by a dedicated BTS7960 H-bridge module rated for 43 A continuous current. We chose these after our initial L298N drivers failed catastrophically during flywheel spin-up, where inrush currents exceeded 20 A per motor.

The BTS7960 modules include built-in thermal shutdown and current sensing. We added external soft-start logic in firmware (ramping PWM over 500 ms) to limit inrush and extend motor brush life.

State Machine Architecture

State Machine & Non-Blocking Launch Sequence

We avoided delay() entirely. The main loop runs continuously: an incremental servo stepper and edge-detected line sensors drive all state transitions without ever blocking. The LAUNCH state sequences flywheels, a timed servo sweep, and a lag timer using only millis() comparisons.

me210_robot.ino — state machine core 
enum State_t {
  INIT,    // Strafe right until center line detected
  FWD,     // Drive forward along the line
  LEFT,    // Correct leftward drift
  RIGHT,   // Correct rightward drift
  LAUNCH,  // Flywheel spin-up → servo sweep → stop
  STOP     // All motors off (terminal)
};

State_t state = INIT;

// Non-blocking servo motion state
unsigned long lastServoStepMs     = 0;
int           servoCurrentAngle   = SERVO_START_ANGLE;
int           servoTargetAngle    = SERVO_START_ANGLE;

// Launch phase timing (all millis()-based, never blocks)
unsigned long launchStartTime     = 0;
unsigned long servoReachedEndTime = 0;
bool          servoSweepStarted   = false;
bool          servoReachedEnd     = false;

void loop() {
  updateServoMotion();   // non-blocking incremental servo step
  checkGlobalEvents();   // line-sensor edge detection → state transitions

  switch (state) {
    case INIT:   strafeRight();  break;
    case FWD:    moveForward();  break;
    case LEFT:   strafeLeft();   break;
    case RIGHT:  strafeRight();  break;

    case LAUNCH:
      stopMoving();
      spinFlywheels();

      // Phase 1: after FLYWHEEL_LEAD_MS (500 ms), begin servo sweep
      if (!servoSweepStarted &&
          (millis() - launchStartTime >= FLYWHEEL_LEAD_MS)) {
        setServoTarget(SERVO_END_ANGLE);
        servoSweepStarted = true;
      }

      // Phase 2: detect when servo reaches its end angle
      if (servoSweepStarted && !servoReachedEnd &&
          isServoAtTarget() &&
          servoTargetAngle == SERVO_END_ANGLE) {
        servoReachedEnd     = true;
        servoReachedEndTime = millis();
      }

      // Phase 3: hold flywheels for FLYWHEEL_LAG_MS (1 s), then stop
      if (servoReachedEnd &&
          (millis() - servoReachedEndTime >= FLYWHEEL_LAG_MS)) {
        resetServoToStart();
        state = STOP;
      }
      break;

    case STOP:
      stopEverything();
      break;
  }
}

// Incremental servo stepper — called every loop(), never blocks
void updateServoMotion() {
  if (millis() - lastServoStepMs < SERVO_STEP_INTERVAL_MS) return;
  lastServoStepMs = millis();
  if (state != LAUNCH) return;

  if (servoCurrentAngle < servoTargetAngle) {
    servoCurrentAngle += SERVO_STEP_SIZE;
    if (servoCurrentAngle > servoTargetAngle)
      servoCurrentAngle = servoTargetAngle;
    launchServo.write(servoCurrentAngle);
  } else if (servoCurrentAngle > servoTargetAngle) {
    servoCurrentAngle -= SERVO_STEP_SIZE;
    if (servoCurrentAngle < servoTargetAngle)
      servoCurrentAngle = servoTargetAngle;
    launchServo.write(servoCurrentAngle);
  }
}

// Edge-detected line sensors drive state transitions
void checkGlobalEvents() {
  if (state == LAUNCH) return;   // freeze line-following during launch
  if (testForCenterBlack())      respToCenterBlack();
  if (testForLeftBlack())        respToLeftBlack();
  if (testForRightBlack())       respToRightBlack();
  if (testForRightEdgeBlack())   respToRightEdgeBlack();
  // ... symmetric white-edge handlers omitted for brevity
}

void respToCenterBlack() {
  if (state == INIT || state == LEFT || state == RIGHT) state = FWD;
}
void respToLeftBlack()  { if (state == FWD) state = LEFT;  }
void respToRightBlack() { if (state == FWD) state = RIGHT; }
void respToRightEdgeBlack() {
  if (state == FWD) enterLaunchState();
}

void enterLaunchState() {
  launchStartTime   = millis();
  servoSweepStarted = false;
  servoReachedEnd   = false;
  stopFlywheels();
  state = LAUNCH;
}

Mecanum Drive & Hardware Helpers

Each mecanum motion primitive sets four IN1/IN2 direction pins then applies per-wheel PWM speeds (individually tuned to compensate for motor variation). Strafing uses a 2× speed multiplier for adequate lateral force. Flywheel and sensor-latch logic rounds out the hardware layer.

me210_robot.ino — mecanum drive & hardware 
// Per-wheel base speeds (tuned per-motor to drive straight)
uint8_t flDriveSpeed = 65;
uint8_t frDriveSpeed = 70;
uint8_t rlDriveSpeed = 65;
uint8_t rrDriveSpeed = 75;

void applyWheelSpeeds() {
  analogWrite(FL_EN, flDriveSpeed);
  analogWrite(FR_EN, frDriveSpeed);
  analogWrite(RL_EN, rlDriveSpeed);
  analogWrite(RR_EN, rrDriveSpeed);
}

void applyWheelSpeeds(uint8_t mul) {
  analogWrite(FL_EN, constrain((int)mul * flDriveSpeed, 0, 255));
  analogWrite(FR_EN, constrain((int)mul * frDriveSpeed, 0, 255));
  analogWrite(RL_EN, constrain((int)mul * rlDriveSpeed, 0, 255));
  analogWrite(RR_EN, constrain((int)mul * rrDriveSpeed, 0, 255));
}

// Mecanum kinematics — each function sets four wheel directions
void moveForward() {
  applyWheelSpeeds();
  digitalWrite(FL_IN1, HIGH); digitalWrite(FL_IN2, LOW);
  digitalWrite(FR_IN1, HIGH); digitalWrite(FR_IN2, LOW);
  digitalWrite(RL_IN1, HIGH); digitalWrite(RL_IN2, LOW);
  digitalWrite(RR_IN1, HIGH); digitalWrite(RR_IN2, LOW);
}

void strafeRight() {
  applyWheelSpeeds(2);   // 2x speed for lateral motion
  digitalWrite(FL_IN1, HIGH); digitalWrite(FL_IN2, LOW);
  digitalWrite(FR_IN1, LOW);  digitalWrite(FR_IN2, HIGH);
  digitalWrite(RL_IN1, LOW);  digitalWrite(RL_IN2, HIGH);
  digitalWrite(RR_IN1, HIGH); digitalWrite(RR_IN2, LOW);
}

void strafeLeft() {
  applyWheelSpeeds(2);
  digitalWrite(FL_IN1, LOW);  digitalWrite(FL_IN2, HIGH);
  digitalWrite(FR_IN1, HIGH); digitalWrite(FR_IN2, LOW);
  digitalWrite(RL_IN1, HIGH); digitalWrite(RL_IN2, LOW);
  digitalWrite(RR_IN1, LOW);  digitalWrite(RR_IN2, HIGH);
}

void stopMoving() {
  digitalWrite(FL_IN1, LOW); digitalWrite(FL_IN2, LOW);
  digitalWrite(FR_IN1, LOW); digitalWrite(FR_IN2, LOW);
  digitalWrite(RL_IN1, LOW); digitalWrite(RL_IN2, LOW);
  digitalWrite(RR_IN1, LOW); digitalWrite(RR_IN2, LOW);
  analogWrite(FL_EN, 0); analogWrite(FR_EN, 0);
  analogWrite(RL_EN, 0); analogWrite(RR_EN, 0);
}

// Flywheel control for the puck launcher
void spinFlywheels() {
  digitalWrite(FLY_L_EN, HIGH);
  digitalWrite(FLY_R_EN, HIGH);
  analogWrite(FLY_LPWM, FLY_TARGET_PWM);  // ~59% duty
}

void stopFlywheels() {
  digitalWrite(FLY_L_EN, LOW);
  digitalWrite(FLY_R_EN, LOW);
  analogWrite(FLY_LPWM, 0);
}

// Edge-latched sensor reads (fires once per black→white transition)
bool isLeftOnBlack = false;

uint8_t testForLeftBlack() {
  if (analogRead(LINE_L) > LINE_L_MIN && !isLeftOnBlack) {
    isLeftOnBlack = true;
    return 1;
  }
  return 0;
}

uint8_t testForLeftWhite() {
  if (analogRead(LINE_L) <= LINE_L_MIN && isLeftOnBlack) {
    isLeftOnBlack = false;
    return 1;
  }
  return 0;
}
